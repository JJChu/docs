    # 滑动窗口中的最大值

    // 我们维护一个大小为 k 的队列
// 依次往队列中放入值，并记录每次的最大值
// 当队列大小为 k 时，我们就要开始把最大值放入到 result 中了
// 这时有个奇妙的现象：在这个队列中，只要最大值存在，其前面的元素就可以移除了，
// 因为它们已经没有意义了，队首永远就是最大的值了
function maxSlidingWindow(nums = [], k = 0) {
  if(!nums.length || !k) return [];
  // 窗口大小为 1，则就是原数组
  if (k === 1) return nums;
  const queue = []; // 窗口
  const res = [];   // 结果
  for (let i = 0; i < nums.length; i++) {
    const num = nums[i];
    if (queue.length >= k) {
      queue.shift();
    }
    while (queue.length && queue[0] <= num) {
      queue.shift();
    }
    queue.push(num);
    if (i >= k - 1) {
      res.push(queue[0]);
    }
  }
  return res;
}

    # 实现一个 Event 模块

    class Event {
  constructor() {
    this.eventMap = {};
  }
  on(event, cb, ...args) {
    cb.args = args;
    (this.eventMap[event] || (this.eventMap[event] = [])).push(cb);
    return this;
  }
  once(event, cb, ...args) {
    const vm = this;

    function temp() {
      vm.off(event, temp);
      cb(...arguments);
    }
    temp.cb = cb;
    vm.on(event, temp, ...args);
    return this;
  }
  fire(event, ...args) {
    (this.eventMap[event] || []).forEach(cb => {
      cb(cb.args.concat(args));
    })
    return this;
  }
  off(event, cb) {
    const cbs = this.eventMap[event] || [];
    if (!cb) {
      this.eventMap[event] = null;
      return this;
    }
    let i = cbs.length;
    while (i--) {
      // cb.cb === cb 这里是手动移除 once 绑定函数
      if (cbs[i] === cb || cbs[i].cb === cb) {
        cbs.splice(i, 1);
      }
    }
    return this;
  }
}

    /**
 * jsonp的实现
 * 返回一个promise 请求成功resolve 失败reject
 */
function jsonp(url, data, opts) {
  return new Promise((resolve, reject) => {
    const id = `jsonp_${Math.random()}`;
    let script = null;
    window[id] = function (data) {
      resolve(data);
      window[id] = null;
      document.head.removeChild(script);
    };
    url += (url.include("?") ? "&" : "?") + data;
    script = document.createElement('script');
    script.src = url;
    script.onerror = function () {
      reject();
      window[id] = null;
      document.head.removeChild(script);
    }
    document.appendChild(script);
  })
}

    # 二分查找

    递归方式

    function binarySearch(arr, val) {
  function search(s, e) {
    const m = Math.ceil((s + e) / 2);
    switch (true) {
      case arr[m] === val:
        return m;
      case s === e:
        return -1;
      case arr[m] < val:
        return search(m, e);
      default:
        return search(s, m - 1);
    }
  }
  return search(0, arr.length - 1);
}

    非递归

    function binarySearch(arr, val) {
  let start = 0;
  let end = arr.length - 1;
  while (start <= end) {
    const middle = Math.ceil((start + end) / 2);
    if (arr[middle] === val) {
      return middle;
    } else if (start === end) {
      return -1;
    } else if(arr[middle] < val) {
      start = middle;
    } else {
      end = middle - 1;
    }
  }
}

    return data;

  const newObj = Array.isArray(data) ? [] : {};

function zip(obj) {
  const result = {};
  function flatJSON(obj, parentKey = "") {
    Object.keys(obj).forEach(key => {
      if (typeof obj[key] === "object") {
        flatJSON(obj[key], `${parentKey}${parentKey ? "." : ""}${key}`)
      } else {
        result[`${parentKey}.${key}`] = obj[key];
      }
    })
  }
  flatJSON(obj);
  return result;
}

// 解压
function unzip(obj) {
  const result = {};
  Object.keys(obj).forEach(key => {
    const keys = key.split(".");
    const val = obj[key];
    keys.reduce((pre, key, i) => {
      !pre[key] && (pre[key] = {});
      if (i === keys.length - 1) {
        pre[key] = val;
      }
      return pre[key]
    }, result);
  })
  return result;
}

  let i = 0;
  return new Promise((resolve, reject) => {
    const next = () => {
      promises[i++]().then(() => {

    /**

 * 
 * 要求如下： 要求最大并发数 maxNum；每当有一个请求返回，就留下一个空位，可以增加新的请求；所有请求完成后，结果按照 urls 里面的顺序依次打出
 * @param {Array} urls

 */
function multiRequest(urls, maxNum) {
  let i = 0
  let promises = []
  return new Promise(r => {
    // 先添加 maxNum 个任务
    while (i < maxNum && i < urls.length) {

        if (i >= urls.length) {

        }
        addTask()

    }

    })
  })
}

    # waterfall 函数

    node async 模块中的 Parallel waterfall

    var tasks = [
  function(callback) {
    callback(null, "one", "two");
  },
  function(arg1, arg2, callback) {
    callback(null, "three");
  },
  function(arg1, callback) {
    callback(null, "done");
  },
]

function waterfall(tasks, callback = NOOP) {
  // 一些边界处理
  if (!Array.isArray(tasks)) return callback(new Error('First argument to waterfall must be an array of functions'));
  if (!tasks.length) return callback();
  var taskIndex = 0;
  // 调用这个函数，则把args传给下一个task，并递归调用自身，直到最后个task
  // 这个函数就是task和task之间的中转，负责处理err和args的传递
  function nextTask(...args) {
    args = args || [];
    // 如果执行完了最后一个task，则调用 callback
    if (taskIndex === tasks.length) {
      return callback.apply(null, [null].concat(args));
    }
    var taskCallback = function(err, ...args) {
      if (err) {
        return callback.apply(null, [err].concat(args));
      }
      nextTask(...args);
    };
    // 把包装后的下一个task作为最后个参数压入当前task的参数栈
    args.push(taskCallback);
    // 取出准备调用的task，taskIndex自增
    var task = tasks[taskIndex++];
    // 调用task
    task.apply(null, args);
  }
  // 调用第一个task
  nextTask();
}

    # Parallel 函数

    node async 模块中的 Parallel waterfall

    var tasks = [
  function(callback) {
    callback(null, "one");
  },
  function(callback) {
    callback(null, "two");
  },
  function(callback) {
    callback(null, "done");
  }
]

function parallel(tasks, resultFn) {
  // 一些边界处理
  if (!Array.isArray(tasks)) return resultFn(new Error('First argument to waterfall must be an array of functions'));
  if (!tasks.length) return resultFn(null, []);
  let results = [];
  let completed = 0;

  function iterator(task, index) {
    function callback(err, ...args){
      if (args.length <= 1) {
        args = args[0];
      }
      results[index] = args;
      if (err) {
        resultFn(err, results);
      } else {
        completed += 1;
        if (completed >= tasks.length) {
          resultFn(null, results);
        }
      }
    }
    task(callback);
  }
  tasks.forEach((task, index) => {
    iterator(task, index);
  });
}

    # 二叉树的最近公共祖先

    解题思路：

    两个节点 p,q 分为两种情况：

    • p 和 q 在相同子树中
    • p 和 q 在不同子树中

    从根节点遍历，递归向左右子树查询节点信息 递归终止条件：如果当前节点为空或等于 p 或 q，则返回当前节点

    • 递归遍历左右子树，如果左右子树查到节点都不为空，则表明 p 和 q 分别在左右子树中，因此，当前节点即为最近公共祖先；
    • 如果左右子树其中一个不为空，则返回非空节点。
    function lowestCommonAncestor(root, p, q) {
  if (!root || root == p || root == q) return root;
  let left = lowestCommonAncestor(root.left, p, q);
  let right = lowestCommonAncestor(root.right, p, q);
  if (left && right) return root;
  return left ? left : right;
}

    const maxPathSum = (root) => {

  const dfs = (root) => {

    const right = dfs(root.right); // 右子树提供的最大收益

    const innerMaxSum = left + root.val + right; // 属于当前子树的最大路径和
    maxSum = Math.max(maxSum, innerMaxSum);      // 挑战一下最大纪录

    const outputMaxSum = root.val + Math.max(left, right); // 对外提供的最大和

    if (outputMaxSum < 0) { // 对外提供的路径和为负，直接返回0
      return 0;
    } else {                // 否则正常返回
      return outputMaxSum;
    }
  };

  dfs(root); // 递归的入口
  return maxSum;
};

    • 先序遍历：根-左-右（1 2 4 5 7 8 3 6）
    • 中序遍历：左-根-右（4 2 7 5 8 1 3 6）

function preOrder(root) {

// 中序遍历
// 对于中序遍历，每当我们到达一个新节点的时候，我们需要考虑：
// 1. 如果他的左节点不是空的，那么我们就不能出栈，而是要继续把他的左节点放进来
// 2. 如果他的左节点是空的，这个时候可以出栈了，但是记得出栈完把自己的右节点放进来。
function inOrder(root) {
  if(!root) return;
  const stack = [];
  while (stack.length || root) {
    if (root) { // 先一直沿左子树向下探索（入栈）
      stack.push(root);
      root = root.left;
    } else {    // 到头后，开始操作（出栈）
      root = stack.pop();
      console.log(root);
      root = root.right;
    }
  }
}

}

    # 归并排序

    归并才用分治思想。主要在两个步骤，拆分+合并。归并的核心就在于 merge 函数，将两个有序数组合并成一个有序数组。

    首先是一种比较好理解的，就是先将数组拆分成两个数组直到不可拆（长度小于 1），然后再开始进行合并。

    function mergeSort(arr) {
  if (arr.length <= 1) return arr;
  const mid = parseInt(arr.length / 2);
  // slice 包括开头，但不包括结尾
  const arr1 = mergeSort(arr.slice(0, mid));
  const arr2 = mergeSort(arr.slice(mid));
  // 开始进行 merge 操作
  return merge(arr1, arr2);
}
function merge(arr1, arr2) {
  const res = [];
  const len1 = arr1.length;
  const len2 = arr2.length;
  let p1 = 0;
  let p2 = 0;
  for (let i = 0; i < (len1 + len2); i++) {
    if (p1 >= len1) {
      res.push(arr2[p2++])
    } else if (p2 >= len2) {
      res.push(arr1[p1++])
    } else if (arr1[p1] < arr2[p2]) {
      res.push(arr1[p1++]);
    } else {
      res.push(arr2[p2++]);
    }
  }
  return res;
}

    但很明显，上面这种方式在空间复杂度上很高，对于排序，有一种“原地替换”的方式可以显著减少空间复杂度

    // 1. 首先改造 merge 函数，把以前“合并两个数组的方式”改成“通过下标在原数组上替换的方式”
// merge 函数由把 arr1 arr2 归并，变成了 arr[start...mid] arr[mid+1...end] 归并
function merge(arr, start, mid, end) {
  // 先把 start...end 的数字暂存
  const temp = [];
  for (let k = start; k <= end; k++) {
    temp[k] = arr[k];
  }
  let p1 = start;
  let p2 = mid + 1;
  for (let i = start; i <= end; i++) {
    if (p1 > mid) {
      arr[i] = temp[p2++];
    } else if (p2 > end) {
      arr[i] = temp[p1++];
    } else if (temp[p1] < temp[p2]) {
      arr[i] = temp[p1++];
    } else {
      arr[i] = temp[p2++];
    }
  }
}
// 剩下的就很简单了，我们把递归“拆分成两个数组”编程“原地拆分”
function mergeSort(arr) {
  sort(arr, 0, arr.length - 1);
}
function sort(arr, start, end) {
  if (start >= end) return;
  const mid = start + parseInt((end - start) / 2);
  sort(arr, start, mid);
  sort(arr, mid + 1, end);
  merge(arr, start, mid, end);
}

    # 快速排序

    快排也是采用的分治思想，只不过采用的是从上而下的思路。快排的关键在于 partition 拆分函数

    • partition 函数首先拿到一个参考值，然后将数组拆分为三部分：arr[start...j-1] <= arr[j] <= arr[j+1...end]
    • 然后再递归排序两边
    function quickSort(arr) {
  sort(arr, 0, arr.length - 1);
}
function sort(arr, start, end) {
  if(start >= end) return;
  // 切分
  const index = partition(arr, start, end);
  sort(arr, start, index - 1);
  sort(arr, index + 1, end);
}
function partition(arr, start, end) {
  let p1 = start, p2 = end + 1; // 左右扫描指针，从两头的再外一层开始
  const v = arr[start];  // 第一个元素作为拆分元素
  while (true) {
    // 左边找到一个大于等于拆分元素的值再停止；
    while (arr[++p1] < v) { if(p1 === end) break; }
    // 右边找到一个小于等于拆分元素的值再停止；
    while (arr[--p2] > v) { if(p2 === start) break; }
    if (p1 >= p2) break;
    exch(arr, p1, p2);
  }
  exch(arr, start, p2);
  return p2;
}
function exch(arr, i, j) {
  const temp = arr[i];
  arr[i] = arr[j]; 
  arr[j] = temp;
}

    我们发现，上面的方式如果有很多重复元素的话，就会做很多无用功。三路快排就能解决这个问题：

    • 需要维护三个指针，lt 左边的都小于切分元素；gt 指针右边的都大于切分元素；i 指针代表当前比较的元素
    • 三路快排会把数组拆分成 arr[start...lt-1] < arr[lt...gt] < arr[gt+1...end]
    function quickSort(arr) {
  sort(arr, 0, arr.length - 1);
}
function sort(arr, start, end) {
  if(start >= end) return;
  const [lt, gt] = quick3way(arr, start, end);
  sort(arr, start, lt - 1);
  sort(arr, gt + 1, end);
}
function quick3way(arr, start, end) {
  let lt = start, i = start + 1, gt = end;
  const v = arr[start];  // 第一个元素作为拆分元素
  while (i <= gt) {
    if (arr[i] < v) {
      exch(arr, lt++, i++)
    } else if (arr[i] > v) {
      exch(arr, i, gt--);
    } else {
      i++
    }
  }
  return [lt, gt];
}
function exch(arr, i, j) {
  const temp = arr[i];
  arr[i] = arr[j]; 
  arr[j] = temp;
}

    function threeSum(nums) {
  if(nums.length < 3) return [];
  const res = [];
  nums.sort((a, b) => a - b);
  for (let i = 0; i < nums.length; i++) {

    let l = i + 1;

    while (l < r) {
      const sum = nums[i] + nums[l] + nums[r];
      if (sum === 0) {
        res.push([nums[i], nums[l], nums[r]]);
        while (l < r && nums[l] === nums[l + 1]) { l++ };
        l++;
        while (l < r && nums[r] === nums[r - 1]) { r-- };
        r--;
      } else if (sum < 0) {
        l++
      } else {
        r--
      }
    }
  }
  return res;
}

    # LRU 缓存

    LRU 要求，

    class LRUCache {
  // 可以给缓存池设置容量限制
  constructor(limit) {
    this.limit = limit;
    this.cache = new Map();
  }
  // set 方法要求
  // 1. 如果池子里有这个 key，则把他挪到最新的位置；
  // 2. 如果没有，超过容量必须删除旧 key，再添加新 key
  set(key, val) {
    if (this.cache.has(key)) {
      this.cache.delete(key);
    } else if (this.cache.size >= this.limit) {
      const oldKey = this.cache.keys().next().value;
      this.cache.delete(oldKey);
    }
    this.cache.set(key, val);
  }
  // get 方法要求
  // 1. 如果没有 key，返回 -1
  // 2. 如果有，返回值，且把他挪到最新的位置
  get(key) {
    if (!this.cache.has(key)) {
      return -1;
    }
    const val = this.cache.get(key);
    this.cache.delete(key);
    this.cache.set(key, val);
    return val;
  }
}

    # K 个一组反转链表

    精选解答

    • 先从链表中抽离出 k 个一组，维护好头和尾；
    • 反转后再和下一组链接
    // 我们先实现一个反转链表的方法
function reverse(head) {
  if (head === null || head.next === null){
    return head;
  }
  let pre = null;
  let cur = head;
  while (cur) {
    [cur.next, pre, cur] = [pre, cur, cur.next];
  }
  return pre;
}
function reverseKGroup(head, k) {
  const dummy = new Node(0, head);
  // 暂存上一组的尾结点
  let preGroupEnd = dummy;
  let curGroupEnd = dummy;
  while (curGroupEnd.next) {
    for (let i = 0; i < k && curGroupEnd !== null; i++) { curGroupEnd = curGroupEnd.next; }
    // 如果 curGroupEnd === null，即需要翻转的链表的节点数小于k，不执行翻转。
    if (curGroupEnd === null) break;
    // 先记录下 curGroupEnd.next，方便后面链接链表
    const nextGroupStart = curGroupEnd.next;
    // 然后断开链表
    curGroupEnd.next = null;
    // 要翻转链表的头节点
    const curGroupStart = preGroupEnd.next;
    // 翻转链表
    preGroupEnd.next = reverse(curGroupStart);
    // 翻转后头节点变到最后（start 和 end 以互换），通过 .next 把断开的链表重新链接
    curGroupStart.next = nextGroupStart;
    // 将 pre 和 end 换成下次要翻转的链表的头结点的上一个节点，重新开始
    preGroupEnd = curGroupStart;
    curGroupEnd = curGroupStart;
  }
  return dummy.next;
}
function Node(value, next) {
  this.value = value;
  this.next = next || null;
}

    // 兑换完留下的瓶盖
    const r = remain % k;
    count += c;
    remain = c + r;
  }
  return count;
}
