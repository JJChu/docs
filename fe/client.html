# 览器中唤起本地 APP

  • URL Schemes
  • iOS/Android 微信及浏览器中唤起本地 APP
  • js在微信、微博、QQ、Safari唤起App的解决方案
  • H5唤起APP指南(附开源唤端库)
  • 关于通过H5页面唤Native户端的介绍
export default function() {
  const ua = navigator.userAgent.toLowerCase(),
    is_android = ua.search("android") > -1,
    from_wechat_or_qq_ua_reg = /(micromessenger|qq)\//.test(ua),
    { location } = window;

  const build_val = url =>
      url ? encodeURIComponent(url === "location" ? location.href : url) : "",
    get_schema_search = opts => {
      const { type } = opts;

      let { val, params } = opts;

      val = build_val(val);
      params = encodeURIComponent(JSON.stringify(params || {}));

      return `?${$.param(
        is_android
          ? {
              from: "h5",
              type,
              val,
              params
            }
          : {
              yq_from: "h5",
              yq_type: type,
              yq_val: val,
              params
            }
      )}`;
    },
    build_sj_download = pkgname =>
      `https://a.app.qq.com/o/simple.jsp?pkgname=${pkgname}`,
    build_schema_for_android = opts =>
      `a17parent://platform.open.api:/parent_main${get_schema_search(opts)}`,
    build_schema_for_ios = opts => `a17parent://jzt${get_schema_search(opts)}`,
    build_download_url = dataset =>
      `https://wx.17zuoye.com/download/17parentapp?cid=${dataset.channel}`,
    build_ios_universal_link = opts =>
      `https://wechat.test.17zuoye.net/j/jzt${get_schema_search(opts)}`;

  const do_open_app = (dataset = {}) => {
    const opts = Object.assign({ type: "webview" }, dataset);

    let schema;

    const download_url = build_download_url(opts),
      sj_download = build_sj_download("com.yiqizuoye.jzt");

    if (is_android) {
      schema = build_schema_for_android(opts);

      if (from_wechat_or_qq_ua_reg) {
        schema = `${sj_download}&android_scheme=${encodeURIComponent(schema)}`;
      }
    } else {
      const ios_version =
        /ip(ad|hone|od)/.test(ua) && ua.match(/os (\d+)_(\d+)/);

      if (ios_version) {
        if (from_wechat_or_qq_ua_reg) {
          schema =
            +ios_version[1] > 8 ? build_ios_universal_link(opts) : sj_download;
        } else {
          schema = build_schema_for_ios(opts);
        }
      } else {
        schema = download_url;
      }
    }

    setTimeout(
      url => {
        location.href = url;
      },
      2200,
      download_url
    );

    location.href = schema;
  };

  $(document).on("click", ".do_open_client", function x() {
    do_open_app($(this).data());
  });

  return do_open_app;
}
