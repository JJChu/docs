# 迭代取值

/**
 * @description: 迭代取值（相当于新语法 obj?.key1?.key2）
 * @param {Object} obj 源数据
 * @param {String} keys 字符串 keys
 * @return 最终匹配到的数据，没有返回 undefined
 *
 * @example
 * const data = { a: { b: [ { c: { d: 1 } } ] } };
 * iteratorGetter(data, "a.b[0]");     // 结果：{ c: { d: 1 } }
 * iteratorGetter(data, "a.b[0].c.d"); // 1
 * iteratorGetter(data, "a.b[0].d");   // undefined
 */
const iteratorGetter = (obj, keys) => {
  if (typeof obj !== "object" || !obj) {
    return obj;
  }
  return keys
    .match(/(\w|\d)+/g)
    .reduce(
      (pre, key) => ([null, undefined].includes(pre) ? pre : pre[key]),
      obj
    );
};

# 迭代赋值

/**
 * @description: 迭代赋值
 * @param {Object} obj 要进行操作的源数据
 * @param {String} keys 字符串 keys
 * @param {Variant} val 最终要赋的值
 * @return 赋好值的数据
 *
 * @example
 * const data = {};
 * iteratorSetter(data, "a.b", 1); // { a: { b: 1 } }
 */
export const iteratorSetter = (obj, keys, val) => {
  if (typeof obj !== "object" || !obj) {
    return obj;
  }
  let result = obj;
  const keyArr = keys.split(".");
  keyArr.forEach((key, index) => {
    if (result[key] === undefined) {
      result[key] = {};
    }
    if (!(result[key] instanceof Object)) {
      throw new TypeError(
        `${keyArr.splice(0, index + 1).join(".")} is not a objcet`
      );
    }
    result = result[key];
    if (index === keyArr.length - 1) {
      result[key] = val;
    }
  });
  return obj;
};

# 合并默认值

怎么去合并默认配置和用户配置，我们通常会有两种方案：

方案一：Object.assgin(defaultOpts, customOpts)

这个方案其实有点小问题，因为是把两个对象合并成一个对象，如果我们删除了某个属性，则就无法拿到默认值了

方案二：Object.create()

这种以原型链的方式能够以更恰当的方式把两个对象结合

const obj = Object.create(defaultOpts);
Object.assgin(obj, customOpts);

# 版本号判断

function compareVersion(v1, v2) {
  v1 = v1.split('.')
  v2 = v2.split('.')
  const len = Math.max(v1.length, v2.length)

  while (v1.length < len) {
    v1.push('0')
  }
  while (v2.length < len) {
    v2.push('0')
  }

  for (let i = 0; i < len; i++) {
    const num1 = parseInt(v1[i])
    const num2 = parseInt(v2[i])

    if (num1 > num2) {
      return 1
    } else if (num1 < num2) {
      return -1
    }
  }

  return 0
}
